CREATE TABLE LEVELS
(
  LEVEL_NO    NUMBER(2),
  LEVEL_NAME  VARCHAR2(20 BYTE)                 NOT NULL,
  STATUS      NUMBER(1),
  PRIMARY KEY
 (LEVEL_NO)
);


CREATE TABLE YEAR
(
  YEAR_NO     NUMBER(3),
  YEAR_NAME   VARCHAR2(20 BYTE)                 NOT NULL,
  START_DATE  DATE,
  END_DATE    DATE,
  IS_CURRENT  VARCHAR2(15 BYTE),
  PRIMARY KEY
 (YEAR_NO)
);


CREATE TABLE RELATION
(
  RELATION_NO    NUMBER(2),
  RELATION_NAME  VARCHAR2(20 BYTE)              NOT NULL,
  STATUS         NUMBER(1),
  PRIMARY KEY
 (RELATION_NO)
);


CREATE TABLE NATIONALITY
(
  NAT_NO    NUMBER(3),
  NAT_NAME  VARCHAR2(30 BYTE)                   NOT NULL,
  STATUS    NUMBER(1),
  PRIMARY KEY
 (NAT_NO)
);


CREATE TABLE JOB
(
  JOB_NO    NUMBER(3),
  JOB_NAME  VARCHAR2(30 BYTE)                   NOT NULL,
  PRIMARY KEY
 (JOB_NO)
);


CREATE TABLE USERS
(
  USER_NO    NUMBER(3),
  USER_NAME  VARCHAR2(30 BYTE)                  NOT NULL,
  PASSWORD   VARCHAR2(30 BYTE),
  FULL_NAME  VARCHAR2(60 BYTE),
  USER_TYPE  NUMBER(1),
  STATUS     NUMBER(1),
  PRIMARY KEY
 (USER_NO)
);


CREATE TABLE CLASSES
(
  CLASS_NO     NUMBER(2),
  CLASS_NAME   VARCHAR2(20 BYTE)                NOT NULL,
  LEVELY       NUMBER(2),
  ORDER_CLASS  NUMBER(5),
  STATUS       NUMBER(1),
  PRIMARY KEY
 (CLASS_NO),
  FOREIGN KEY (LEVELY) 
 REFERENCES LEVELS (LEVEL_NO)
);


CREATE TABLE STUDENT
(
  STUD_NO           NUMBER(5),
  STUD_NAME         VARCHAR2(50 BYTE)           NOT NULL,
  AGE               NUMBER(3),
  ADRESS            VARCHAR2(30 BYTE),
  SEX               NUMBER(1),
  BRITH_DATE        DATE,
  BRITH_PLACE       VARCHAR2(20 BYTE),
  CLASS             NUMBER(2),
  NATIONALITY       NUMBER(3),
  PARENT_NAME       VARCHAR2(50 BYTE),
  PARENT_JOB        NUMBER(3),
  PARENT_PHONE      VARCHAR2(20 BYTE),
  PARENT_WORKPHONE  VARCHAR2(20 BYTE),
  JOB_ADRESS        VARCHAR2(30 BYTE),
  RELATION          NUMBER(2),
  LEVELY            NUMBER(2),
  STATUS            NUMBER(1),
  PRIMARY KEY
 (STUD_NO),
  FOREIGN KEY (LEVELY) 
 REFERENCES LEVELS (LEVEL_NO),
  FOREIGN KEY (RELATION) 
 REFERENCES RELATION (RELATION_NO),
  FOREIGN KEY (PARENT_JOB) 
 REFERENCES JOB (JOB_NO),
  FOREIGN KEY (NATIONALITY) 
 REFERENCES NATIONALITY (NAT_NO),
  FOREIGN KEY (CLASS) 
 REFERENCES CLASSES (CLASS_NO)
);


CREATE TABLE SUBJECT
(
  SUB_NO    NUMBER(2),
  SUB_NAME  VARCHAR2(20 BYTE)                   NOT NULL,
  CLASS     NUMBER(2),
  STATUS    NUMBER(1),
  PRIMARY KEY
 (SUB_NO),
  FOREIGN KEY (CLASS) 
 REFERENCES CLASSES (CLASS_NO)
);


CREATE TABLE SHOBA
(
  SHOBA_NO    NUMBER(2),
  SHOBA_NAME  VARCHAR2(20 BYTE)                 NOT NULL,
  CLASS       NUMBER(2),
  STATUS      NUMBER(1),
  PRIMARY KEY (SHOBA_NO,class),
  FOREIGN KEY (CLASS) 
 REFERENCES CLASSES (CLASS_NO)
);


CREATE TABLE DEGREE
(
  STUDENT  NUMBER(5),
  SUBJECT  NUMBER(2),
  MARK1    NUMBER(3),
  MARK2    NUMBER(3),
  MARK3    NUMBER(3),
  MARK4    NUMBER(3),
  MARK5    NUMBER(3),
  FOREIGN KEY (SUBJECT) 
 REFERENCES SUBJECT (SUB_NO),
  FOREIGN KEY (STUDENT) 
 REFERENCES STUDENT (STUD_NO)
);


CREATE TABLE YEAR_STUD
(
  REGISTER  NUMBER(5),
  STUDENT   NUMBER(5),
  CLASS     NUMBER(2),
  SHOBA     NUMBER(2),
  LEVELY    NUMBER(5),
  YEAR      NUMBER(3),
  PRIMARY KEY
 (REGISTER),
  FOREIGN KEY (YEAR) 
 REFERENCES YEAR (YEAR_NO),
  FOREIGN KEY (LEVELY) 
 REFERENCES LEVELS (LEVEL_NO),
  FOREIGN KEY (SHOBA) 
 REFERENCES SHOBA (SHOBA_NO),
  FOREIGN KEY (CLASS) 
 REFERENCES CLASSES (CLASS_NO),
  FOREIGN KEY (STUDENT) 
 REFERENCES STUDENT (STUD_NO)
);
